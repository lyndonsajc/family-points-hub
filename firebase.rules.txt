Firestore Rules - simple family app testing version

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read, write: if true;
    }
  }
}

Storage Rules - simple family app testing version

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if true;
    }
  }
}

IMPORTANT:
These rules are open for simple family testing.
For a private app, add Firebase Authentication later.
